<script>
let items = JSON.parse(localStorage.getItem("items")) || [];
let transactions = JSON.parse(localStorage.getItem("transactions")) || [];

function save() {
localStorage.setItem("items", JSON.stringify(items));
localStorage.setItem("transactions", JSON.stringify(transactions));
}

function render() {
let table = document.querySelector("#stockTable tbody");
let select = document.getElementById("itemSelect");

table.innerHTML = "";
select.innerHTML = "";

items.forEach((item, index) => {
table.innerHTML += `
<tr>
<td>${item.name}</td>
<td>${item.stock}</td>
<td>
<button onclick="quickWithdraw(${index})">-1</button>
</td>
</tr>
`;

select.innerHTML += `<option value="${index}">${item.name}</option>`;
});

updateDashboard();
}

function addItem() {
let name = document.getElementById("newItem").value;
if (!name) return;
items.push({ name, stock: 0 });
save();
render();
}

function withdraw() {
let i = itemSelect.value;
let qty = parseInt(document.getElementById("qty").value);
let user = document.getElementById("user").value;

if (!qty || !user) return;

items[i].stock -= qty;

transactions.push({
type: "withdraw",
item: items[i].name,
qty,
user,
date: new Date()
});

save();
render();
}

function addStock() {
let i = itemSelect.value;
let qty = parseInt(document.getElementById("qty").value);
let user = document.getElementById("user").value;

if (!qty || !user) return;

items[i].stock += qty;

transactions.push({
type: "add",
item: items[i].name,
qty,
user,
date: new Date()
});

save();
render();
}

function quickWithdraw(i) {
items[i].stock -= 1;

transactions.push({
type: "withdraw",
item: items[i].name,
qty: 1,
user: "quick",
date: new Date()
});

save();
render();
}

function updateDashboard() {
let totalWithdraw = transactions
.filter(t => t.type === "withdraw")
.reduce((sum, t) => sum + t.qty, 0);

document.getElementById("summary").innerText =
"ยอดเบิกรวม: " + totalWithdraw;
}

render();
</script>
